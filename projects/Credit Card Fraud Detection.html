def model_evaluate(y_test,y_pred):
    accuracy=accuracy_score(y_test, y_pred)
    F1_score=f1_score(y_test, y_pred, average=None)
    print('Accuracy is {}'.format(round(accuracy, 3)))
    print('F1 score is:', round(F1_score.mean(), 3))
